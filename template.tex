\newcommand{\headerlessSection}[1]{
  \par\refstepcounter{section}
  \sectionmark{#1}
  \addcontentsline{toc}{section}{\protect\numberline{\thesection}#1}
}

\makeatletter
\def\g@addto@macroname@exp#1#2{%
  \expandafter\g@addto@macro\csname#1\expandafter\endcsname\expandafter{#2}}

\NewEnviron{rgrp}[2][]{%
  \if!#1!\def\rgrp@key{#2}\else\def\rgrp@key{#1}\fi
  \global\@namedef{thergrp@\rgrp@key}{\headerlessSection{#2}}
  \g@addto@macroname@exp{thergrp@\rgrp@key}{\BODY}%
}

\def\rtitle #1{
	\vspace{14pt}
	\textbf{\uppercase{#1}}
	\vspace{4pt}
}

\def\rinterests #1{
	\vspace{12pt}
	{\large \textbf{\uppercase{Interests:}}} #1
}

\def\redu #1#2#3#4#5{
	{\large \textbf{#1}
	\hfill
	\textbf{#2}}
	\vspace{2pt}
	\newline
	#3
	\hfill
	#4
	\vspace{2pt}
	\newline
	{\large \textbf{Coursework:} #5}
	\vspace{4pt}
}

\def \rskills #1#2#3{
	\textbf{Proficient:} #1
	\hfill
	\textbf{Familiar:} #2
	\hfill
	\newline
	\textbf{Working Knowledge:} #3
}

\def\rextrainfo[#1]#2#3{
	{\large 
	\textbf{#2}
	\hfill
	\textbf{#3}
	}
	\begin{itemize}
	\foreach \x in {#1}{
		\item \x
	}
	\end{itemize}
	\vspace{2pt}
}

\def\rextra #1#2{
	{\large 
	\textbf{#1}
	\hfill
	#2
	}
	\newline
}

\def\rexp[#1]#2#3{
	{\large 
	\textbf{#2}
	\hfill
	\textbf{#3}
	}
	\begin{itemize}
	\foreach \x in {#1}{
		\item \x
	}
	\end{itemize}
	\vspace{2pt}
}

\def\rproj[#1]#2#3#4#5{
	{\large
	\textbf{#2 \href{#5}{\underline{\smash{#4}}}
	\hfill
	\textbf{#3}}
	}
	\begin{itemize}
	\foreach \x in {#1}{
		\item \x
	}
	\end{itemize}
}
